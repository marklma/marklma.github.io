文件夹加密小程序

十一月 29 2008 | 评论

用户手册：本程序只能隐藏，不能防止文件夹被删除运行本程序后会生成：加解密.bat、readme.txt 只需运行 加解密.bat输入密码即可解密，再运行 加解密.bat直接确认即可加密 VISTA下本程序不稳定，需用户自己重命名加密后的文件夹才可若重命名加密文件夹，需删除原来的 加解密.bat 自己重新用本程序生成新文件夹名应的 加解密.bat 请将以下代码复制到记事本中，然后该名成 请将其放入你想生成隐藏文件夹的目录下.bat 运行即可

@echo off
echo         文件加密小程序
echo 本程序能在WINDOWS系统下生成一个新文件夹
echo 并让该文件夹变成回收站并隐藏而起到加密作用
echo 若不用以上方法，还有 方法二：(非VISTA系统下)
echo 直接在您的文件夹名字后加上.{645FF040-5081-101B-9F08-00AA002F954E}
echo 用方法二加密若想恢复直接更改文件夹名字即可
echo.
echo 请按任意键开始用方法一加密，否则请关闭
pause >nul
cls
set filename=autorun.inf
set password=m
set/p filename=请输入想生成的文件夹的名字,(默认文件夹名称为autorun.inf):
set/p password=请输入能进行解密的密码(默认密码为m):
echo 请稍后...
echo @echo off >加解密.bat
echo set/p ma=>>加解密.bat
echo if "%%ma%%"=="%password%" (goto d) >>加解密.bat
echo @echo 请稍后... >>加解密.bat
echo attrib -s -h -r "%filename%" /s /d >> 加解密.bat
echo echo [.ShellClassInfo] ^> "%filename%\desktop.ini" >> 加解密.bat
echo echo CLSID={645FF040-5081-101B-9F08-00AA002F954E} ^>^> "%filename%\desktop.ini" >> 加解密.bat
echo attrib +h +r +s "%filename%\desktop.ini" >> 加解密.bat
echo attrib +r "%filename%" /s /d >> 加解密.bat
echo attrib +h +s "%filename%" /s /d >> 加解密.bat
echo goto end >> 加解密.bat
echo :d >> 加解密.bat
echo echo 请稍后... >>加解密.bat
echo attrib -h -s -r "%filename%" /s /d >> 加解密.bat
echo attrib -s -h "%filename%\desktop.ini" >> 加解密.bat
echo del /f /q "%filename%\desktop.ini" >> 加解密.bat
echo :end >> 加解密.bat
echo echo OK! >> 加解密.bat
md "%filename%"
echo OK!
echo.
echo ！！！        注意：            ！！！
echo 若要加密请运行 加解密.bat ，然后直接点回车
echo 若要解密请运行 加解密.bat ，输入密码后回车
echo         不要重命名加密文件夹，除非
echo VISTA下显示隐藏文件发现它没变成回收站样式需自己重命名文件夹名
echo.
echo                         作者：Mark.L.Ma
echo                         e-mail：4258882@163.com
echo.
echo 阅读完注意并放文件进文件夹后，请按确认开始加密
attrib +r +s 加解密.bat
call 加解密.bat
del /f /q /a attrib.exe >nul
del /f /q /a %0 && readme.txt
